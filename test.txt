local NotificationLibrary = {}
local notifications = {}
local notificationHeight = 85
local notificationWidth = 330
local notificationPadding = 10
local notificationTransparency = 0.3
local notificationImage = true
local notificationImageUiCorner = 12
local closeButtonSize = UDim2.new(0, 20, 0, 20)
local closeButtonOffset = UDim2.new(1, -24, 0, 0.5)
local notificationUiStroke = true
local notificationUiStrokeTransparency = 0.5
local notificationUiStrokeThickness = 2
local notificationAppearance = "left"
local showCountdown = false
local notificationGuiType = "CG"

local screenGui
if notificationGuiType == "SG" then
    screenGui = Instance.new("ScreenGui")
    screenGui.ResetOnSpawn = false
    screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")
elseif notificationGuiType == "CG" then
    screenGui = Instance.new("ScreenGui")
    screenGui.ResetOnSpawn = false
    screenGui.Parent = game:GetService("CoreGui")
end

local notificationSound = Instance.new("Sound")
notificationSound.SoundId = "rbxassetid://7383525713"
notificationSound.Volume = 2
notificationSound.Parent = screenGui

local closeButtonSound = Instance.new("Sound")
closeButtonSound.SoundId = "rbxassetid://17208361335"
closeButtonSound.Volume = 0
closeButtonSound.Parent = screenGui

local function updateNotificationPositions(callback)
    local moveDuration = 0.3
    for i, notificationFrame in ipairs(notifications) do
        local targetPosition = UDim2.new(
            notificationAppearance == "left" and 0 or 1, 
            notificationAppearance == "left" and 10 or -notificationWidth - 10,
            1, 
            -(i * (notificationHeight + notificationPadding))
        )

        local moveTween = TweenService:Create(notificationFrame, TweenInfo.new(moveDuration, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
            Position = targetPosition
        })
        moveTween:Play()

        if i == #notifications and callback then
            moveTween.Completed:Connect(callback)
        end
    end
end

function NotificationLibrary:CreateNotification(title, description, duration, image)
    duration = duration or 3
    local remainingTime = duration

    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, notificationWidth, 0, notificationHeight)

    if notificationAppearance == "left" then
        frame.Position = UDim2.new(0, -notificationWidth - 10, 1, 0)
    elseif notificationAppearance == "right" then
        frame.Position = UDim2.new(1, notificationWidth + 10, 1, 0)
    end

    frame.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    frame.BackgroundTransparency = notificationTransparency
    frame.BorderSizePixel = 0
    frame.Parent = screenGui

    if notificationUiStroke then
        local uiStroke = Instance.new("UIStroke")
        uiStroke.Thickness = notificationUiStrokeThickness
        uiStroke.Transparency = notificationUiStrokeTransparency
        uiStroke.Color = Color3.fromRGB(255, 255, 255)
        uiStroke.Parent = frame
    end

    local uiCorner = Instance.new("UICorner")
    uiCorner.CornerRadius = UDim.new(0, 12)
    uiCorner.Parent = frame
    local contentFrame = Instance.new("Frame")
    contentFrame.Size = notificationImage and image
        and UDim2.new(1, -notificationHeight - 5, 1, 0)
        or UDim2.new(1, -10, 1, 0)
    contentFrame.Position = notificationImage and image
        and UDim2.new(0, notificationHeight + 2, 0, 0)
        or UDim2.new(0, 10, 0, 0)
    contentFrame.BackgroundTransparency = 1
    contentFrame.Parent = frame

    local titleLabel = Instance.new("TextLabel")
    titleLabel.Size = UDim2.new(1, 0, 0, 20)
    titleLabel.Position = UDim2.new(0, 0, 0, 0)
    titleLabel.BackgroundTransparency = 1
    titleLabel.Text = title
    titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    titleLabel.TextScaled = true
    titleLabel.TextXAlignment = Enum.TextXAlignment.Left
    titleLabel.Font = Enum.Font.GothamBold
    titleLabel.Parent = contentFrame

    local divider = Instance.new("Frame")
    divider.Size = notificationImage and image
        and UDim2.new(1.41, -(notificationHeight + 10), 0, 1)
        or UDim2.new(1, 0, 0, 1)
    divider.Position = UDim2.new(0, 0, 0, 20)
    divider.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
    divider.BackgroundTransparency = 0.3
    divider.BorderSizePixel = 0
    divider.Parent = contentFrame

    local descriptionLabel = Instance.new("TextLabel")
    descriptionLabel.Size = UDim2.new(1, 0, 1, -21)
    descriptionLabel.Position = UDim2.new(0, 0, 0, 21)
    descriptionLabel.BackgroundTransparency = 1
    descriptionLabel.Text = description
    descriptionLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    descriptionLabel.TextScaled = true
    descriptionLabel.TextWrapped = true
    descriptionLabel.TextXAlignment = Enum.TextXAlignment.Left
    descriptionLabel.TextYAlignment = Enum.TextYAlignment.Top
    descriptionLabel.Font = Enum.Font.Gotham
    descriptionLabel.ClipsDescendants = true
    descriptionLabel.Parent = contentFrame

    if notificationImage and image then
        local imageLabel = Instance.new("ImageLabel")
        imageLabel.Size = UDim2.new(0, notificationHeight - 5, 0, notificationHeight - 5)
        imageLabel.Position = UDim2.new(0, 3, 0.5, -(notificationHeight - 5) / 2)
        imageLabel.BackgroundTransparency = 1
        imageLabel.Image = image
        imageLabel.ScaleType = Enum.ScaleType.Fit
        imageLabel.Parent = frame

        if notificationImageUiCorner > 0 then
            local imageUiCorner = Instance.new("UICorner")
            imageUiCorner.CornerRadius = UDim.new(0, notificationImageUiCorner)
            imageUiCorner.Parent = imageLabel
        end
    end

    local closeButton = Instance.new("ImageButton")
    closeButton.Size = closeButtonSize
    closeButton.Position = closeButtonOffset
    closeButton.BackgroundTransparency = 1
    closeButton.Image = "rbxassetid://108386127004478"
    closeButton.Parent = frame

    local function closeNotification()
        closeButtonSound:Play()
        local slideOutTween
        if notificationAppearance == "left" then
            slideOutTween = TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Position = UDim2.new(0, -notificationWidth - 10, 1, frame.Position.Y.Offset)})
        elseif notificationAppearance == "right" then
            slideOutTween = TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Position = UDim2.new(1, notificationWidth + 10, 1, frame.Position.Y.Offset)})
        end

        slideOutTween:Play()

        slideOutTween.Completed:Connect(function()
            for i, v in ipairs(notifications) do
                if v == frame then
                    table.remove(notifications, i)
                    break
                end
            end
            frame:Destroy()
            updateNotificationPositions()
        end)
    end

    closeButton.MouseButton1Click:Connect(closeNotification)
    table.insert(notifications, 1, frame)

    notificationSound:Play()

    updateNotificationPositions(function()
        local slideInTween
        local targetPosition = UDim2.new(
            notificationAppearance == "left" and 0 or 1, 
            notificationAppearance == "left" and 10 or -notificationWidth - 10,
            1, 
            -(1 * (notificationHeight + notificationPadding))
        )
        slideInTween = TweenService:Create(frame, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Position = targetPosition})

        slideInTween:Play()

        slideInTween.Completed:Connect(function()
            delay(duration, closeNotification)
        end)
    end)
end

local function StopAnim(plr)
    local success, errorMsg = pcall(function()
        if not plr.Character or not plr.Character:FindFirstChildOfClass("Humanoid") then
            return
        end

        local hum = plr.Character:FindFirstChildOfClass("Humanoid")
        local animate = plr.Character:FindFirstChild("Animate")
        if animate then
            animate.Disabled = true
        end

        for _, track in ipairs(hum:GetPlayingAnimationTracks()) do
            track:Stop()
        end
    end)

    if not success then
        warn("Failed to stop animation: " .. errorMsg)
    end
end

local function PlayAnim(plr, id, time, speed)
    local success, errorMsg = pcall(function()
        if not plr.Character or not plr.Character:FindFirstChildOfClass("Humanoid") then
            return
        end

        local hum = plr.Character:FindFirstChildOfClass("Humanoid")

        StopAnim(plr)

        local anim = Instance.new("Animation")
        anim.AnimationId = "rbxassetid://" .. id
        local animTrack = hum:LoadAnimation(anim)
        animTrack:Play()
        animTrack.TimePosition = time
        animTrack:AdjustSpeed(speed)

        animTrack.Stopped:Connect(function()
            StopAnim(plr)
            local animate = plr.Character:FindFirstChild("Animate")
            if animate then
                animate.Disabled = false
            end
        end)
    end)

    if not success then
        warn("fail to change animation: " .. errorMsg)
    end
end

clickSound.Parent = Toggle
clickSound.Volume = 2

local dragging
local dragInput
local dragStart
local startPos

local function update(input)
    local delta = input.Position - dragStart
    local newPos = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    local tween = TweenService:Create(Toggle, TweenInfo.new(0.5, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Position = newPos})
    tween:Play()
end

Toggle.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = Toggle.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

Toggle.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input == dragInput then
        update(input)
    end
end)

local function playClickAnimation()
    local shrinkTween = TweenService:Create(Toggle, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(0, 55, 0, 55)})
    local growTween = TweenService:Create(Toggle, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.new(0, 60, 0, 60)})
    local colorTween = TweenService:Create(Toggle, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageColor3 = Color3.fromRGB(90, 90, 90)})
    local restoreColorTween = TweenService:Create(Toggle, TweenInfo.new(0.1, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageColor3 = Color3.fromRGB(255, 255, 255)})

    shrinkTween:Play()
    shrinkTween.Completed:Connect(function()
        growTween:Play()
    end)
    
    colorTween:Play()
    colorTween.Completed:Connect(function()
        restoreColorTween:Play()
    end)
end

Toggle.MouseButton1Click:Connect(function()
    playClickAnimation()
    clickSound:Play()
    if Library and Library.ToggleUI then
        Library:ToggleUI()
    else
        warn("Library or Library:ToggleUI() is not defined.")
    end
end)
